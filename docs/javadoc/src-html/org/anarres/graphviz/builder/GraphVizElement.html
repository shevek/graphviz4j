005package org.anarres.graphviz.builder;
006
007import javax.annotation.CheckForNull;
008import javax.annotation.Nonnull;
009
010/**
011 *
012 * @author shevek
013 */
014public abstract class GraphVizElement<T extends GraphVizElement<?>> {
015
016    @Nonnull
017    private final GraphVizGraph graph;
018    @CheckForNull
019    private GraphVizLabel label;
020    @CheckForNull
021    private String color;
022    @CheckForNull
023    private String style;
024
025    /* pp */ GraphVizElement(@Nonnull GraphVizGraph graph) {
026        this.graph = graph;
027    }
028
029    public GraphVizGraph getGraph() {
030        return graph;
031    }
032
033    @CheckForNull
034    public GraphVizLabel getLabel() {
035        return label;
036    }
037
038    @Nonnull
039    public GraphVizLabel label() {
040        if (label == null)
041            label = new GraphVizLabel();
042        return label;
043    }
044
045    @Nonnull
046    @SuppressWarnings("unchecked")
047    public T label(@Nonnull CharSequence csq) {
048        label().set(csq);
049        return (T) this;
050    }
051
052    @CheckForNull
053    public String getColor() {
054        return color;
055    }
056
057    @Nonnull
058    @SuppressWarnings("unchecked")
059    public T color(@Nonnull String color) {
060        this.color = color;
061        return (T) this;
062    }
063
064    @CheckForNull
065    public String getStyle() {
066        return style;
067    }
068
069    @Nonnull
070    @SuppressWarnings("unchecked")
071    public T style(@Nonnull String style) {
072        this.style = style;
073        return (T) this;
074    }
075}
