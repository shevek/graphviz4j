005package org.anarres.graphviz.builder;
006
007import com.google.common.base.Objects;
008import com.google.common.base.Preconditions;
009import javax.annotation.Nonnull;
010
011/**
012 *
013 * @author shevek
014 */
015public abstract class GraphVizObject<T extends GraphVizObject<?>> extends GraphVizElement<T> {
016
017    /* pp */ static class Key {
018
019        @Nonnull
020        private final GraphVizScope scope;
021        @Nonnull
022        private final Object object;
023
024        public Key(@Nonnull GraphVizScope scope, @Nonnull Object object) {
025            this.scope = Preconditions.checkNotNull(scope, "Scope was null.");
026            this.object = Preconditions.checkNotNull(object, "Object was null.");
027        }
028
029        @Nonnull
030        public GraphVizScope getScope() {
031            return scope;
032        }
033
034        @Nonnull
035        public Object getObject() {
036            return object;
037        }
038
039        @Override
040        public int hashCode() {
041            return System.identityHashCode(getScope()) ^ getObject().hashCode();
042        }
043
044        @Override
045        public boolean equals(Object obj) {
046            if (this == obj)
047                return true;
048            if (obj == null)
049                return false;
050            if (getClass() != obj.getClass())
051                return false;
052            Key k = (Key) obj;
053            return getScope() == k.getScope()
054                    && getObject().equals(k.getObject());
055        }
056
057        @Override
058        public String toString() {
059            return Objects.toStringHelper(this)
060                    .add("scope", scope.getClass().getSimpleName() + "@" + System.identityHashCode(scope))
061                    .add("object", object.getClass().getSimpleName() + "@" + System.identityHashCode(object))
062                    .toString();
063        }
064
065    }
066    @Nonnull
067    private final Key key;
068    @Nonnull
069    private final String id;
070
071    /* pp */ GraphVizObject(@Nonnull GraphVizGraph graph, @Nonnull Key key, @Nonnull String id) {
072        super(graph);
073        this.key = key;
074        this.id = id;
075    }
076
077    @Nonnull
078    /* pp */ Key getKey() {
079        return key;
080    }
081
082    @Nonnull
083    public GraphVizScope getScope() {
084        return getKey().getScope();
085    }
086
087    @Nonnull
088    /* pp */ String getId() {
089        return id;
090    }
091}
