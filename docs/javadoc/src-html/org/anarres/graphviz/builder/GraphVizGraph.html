079    private void clear(@Nonnull GraphVizScope scope, @Nonnull Iterable<? extends GraphVizObject.Key> data) {
080        Iterator<? extends GraphVizObject.Key> it = data.iterator();
081        while (it.hasNext()) {
082            GraphVizObject.Key key = it.next();
083            if (Objects.equal(key.getScope(), scope))
084                it.remove();
085        }
086    }
087
088    public void clear(@Nonnull GraphVizScope scope) {
089        clear(scope, clusters.keySet());
090        for (GraphVizCluster cluster : clusters.values()) {
091            clear(scope, cluster.getClusterKeys());
092            clear(scope, cluster.getNodeKeys());
093        }
094
095        clear(scope, nodes.keySet());
096
097        Iterator<? extends GraphVizEdge.Key> it = edges.keySet().iterator();
098        while (it.hasNext()) {
099            GraphVizEdge.Key key = it.next();
100            if (Objects.equal(key.getSourceKey().getScope(), scope))
101                it.remove();
102            else if (Objects.equal(key.getTargetKey().getScope(), scope))
103                it.remove();
104        }
105    }

107    @Nonnull
108    public GraphVizNode node(@Nonnull GraphVizScope scope, @Nonnull Object object) {
109        GraphVizNode.Key key = new GraphVizNode.Key(scope, object);
110        if (!isScopeVisible(scope))
111            return new GraphVizNode(this, key, -1);
112        GraphVizNode node = nodes.get(key);
113        if (node == null) {
114            int id = counter++;
115            // LOG.info("n" + id + " -> " + object, new Exception());
116            node = new GraphVizNode(this, key, id);
117            nodes.put(key, node);
118        }
119        return node;
120    }
121
122    public boolean containsNode(@Nonnull GraphVizScope scope, @Nonnull Object object) {
123        GraphVizNode.Key key = new GraphVizNode.Key(scope, object);
124        return nodes.containsKey(key);
125    }

127    @Nonnull
128    public GraphVizEdge edge(@Nonnull GraphVizObject<?> source, @Nonnull GraphVizObject<?> target) {
129        GraphVizEdge.Key key = new GraphVizEdge.Key(source.getKey(), target.getKey());
130        if (!isScopeVisible(source.getScope()))
131            return new GraphVizEdge(this, key, "", "");
132        if (!isScopeVisible(target.getScope()))
133            return new GraphVizEdge(this, key, "", "");
134        GraphVizEdge edge = edges.get(key);
135        if (edge == null) {
136            edge = new GraphVizEdge(this, key, source.getId(), target.getId());
137            edges.put(key, edge);
138            if (source instanceof GraphVizCluster)
139                edge.logicalTail(source.getId());
140            if (target instanceof GraphVizCluster)
141                edge.logicalHead(target.getId());
142        }
143        return edge;
144    }

146    @Nonnull
147    public GraphVizEdge edge(@Nonnull GraphVizScope scope, @Nonnull Object source, @Nonnull Object target) {
148        return edge(node(scope, source), node(scope, target));
149    }

151    public boolean containsEdge(@Nonnull GraphVizObject<?> source, @Nonnull GraphVizObject<?> target) {
152        GraphVizEdge.Key key = new GraphVizEdge.Key(source.getKey(), target.getKey());
153        return edges.containsKey(key);
154    }

156    public boolean containsEdge(@Nonnull GraphVizScope scope, @Nonnull Object source, @Nonnull Object target) {
157        GraphVizObject.Key k_source = new GraphVizObject.Key(scope, source);
158        GraphVizObject.Key k_target = new GraphVizObject.Key(scope, target);
159        GraphVizEdge.Key key = new GraphVizEdge.Key(k_source, k_target);
160        return edges.containsKey(key);
161    }

163    @Nonnull
164    private GraphVizCluster cluster(@CheckForNull GraphVizCluster parent, @Nonnull GraphVizScope scope, @Nonnull Object object) {
165        GraphVizCluster.Key key = new GraphVizCluster.Key(scope, object);
166        if (!isScopeVisible(scope))
167            return new GraphVizCluster(this, key, null, -1);
168        GraphVizCluster cluster = clusters.get(key);
169        if (cluster == null) {
170            cluster = new GraphVizCluster(this, key, parent, counter++);
171            clusters.put(key, cluster);
172        }
173        return cluster;
174    }

176    @Nonnull
177    public GraphVizCluster cluster(@Nonnull GraphVizScope scope, @Nonnull Object object) {
178        return cluster(null, scope, object);
179    }

183        return cluster(parent, parent.getScope(), object);

186    private static boolean append(@Nonnull Appendable writer, @Nonnull String key, @CheckForNull Object value, boolean quote, boolean first) throws IOException {
187        if (value == null)
188            return first;
189        if (!first)
190            writer.append(',');
191        writer.append(key).append('=');
192        if (quote)
193            writer.append('"');
194        writer.append(String.valueOf(value));
195        if (quote)
196            writer.append('"');
197        return false;
198    }

200    private static void writeIndent(@Nonnull Writer writer, @Nonnegative int depth) throws IOException {
201        for (int i = 0; i < depth; i++)
202            writer.write('\t');
203    }
204
205    private void writeTo(
206            @Nonnull Writer writer,
207            @Nonnull Map<? extends GraphVizCluster, ? extends Iterable<? extends GraphVizCluster>> clusterMap,
208            @CheckForNull GraphVizCluster parent,
209            @Nonnegative int depth) throws IOException {
210        Iterable<? extends GraphVizCluster> _clusters = clusterMap.remove(parent);
211        if (_clusters == null)
212            return;
213        for (GraphVizCluster cluster : _clusters) {
214            writeIndent(writer, depth);
215            writer.append("\tsubgraph ").append(cluster.getId()).append(" {\n");
216            writeIndent(writer, depth);
217            writer.append("\t\t// scope=").append(String.valueOf(System.identityHashCode(cluster.getKey().getScope()))).append("\n");
218            writeIndent(writer, depth);
219            writer.append("\t\t").append(cluster.getId()).append(" [label=\"\",shape=point,style=invis];\n");
220            if (cluster.getLabel() != null) {
221                writeIndent(writer, depth);
222                writer.append("\t\t").append("label=\"").append(String.valueOf(cluster.getLabel())).append("\";\n");
223            }
224            for (GraphVizNode.Key key : cluster.getNodeKeys()) {
225                GraphVizNode node = node(key.getScope(), key.getObject());
226                writeIndent(writer, depth);
227                writer.append("\t\t").append(node.getId()).append(";\n");
228            }
229            writeTo(writer, clusterMap, cluster, depth + 1);
230            writeIndent(writer, depth);
231            writer.append("\t}\n");
232        }
233    }

235    /** This map allows null keys, and the null key should be present. */
236    @Nonnull
237    private Map<GraphVizCluster, List<GraphVizCluster>> newClusterMap() {
238        Map<GraphVizCluster, List<GraphVizCluster>> clusterMap = new HashMap<GraphVizCluster, List<GraphVizCluster>>();
239        for (GraphVizCluster child : clusters.values()) {
240            @CheckForNull
241            GraphVizCluster parent = child.getParent();
242            List<GraphVizCluster> children = clusterMap.get(parent);
243            if (children == null) {
244                children = new ArrayList<GraphVizCluster>();
245                clusterMap.put(parent, children);
246            }
247            children.add(child);
248        }
249        return clusterMap;
250    }

252    public void writeTo(@Nonnull Writer out) throws IOException {
253        Writer writer;
254        if (out instanceof BufferedWriter)
255            writer = out;
256        else if (out instanceof StringWriter)
257            writer = out;
258        // else if (out instanceof StringBuilderWriter) writer = out;
259        else
260            writer = new BufferedWriter(out);
261        writer.write("digraph G {\n");
262        writer.write("\tcompound=true;\n");
263        // writer.write("\tranksep=1.5;\n");
264        writer.write("\tnode [shape=box];\n");
265        if (getLabel() != null) {
266            writer.append("\t");
267            append(writer, "label", getLabel(), true, true);
268            writer.append(";\n");
269        }
270
271        for (GraphVizNode node : nodes.values()) {
272            writer.append("\t").append(node.getId()).append(" [");
273            boolean first = true;
274            first = append(writer, "color", node.getColor(), true, first);
275            first = append(writer, "style", node.getStyle(), true, first);
276            first = append(writer, "shape", node.getShape(), true, first);
277            first = append(writer, "label", node.getLabel(), true, first);
278            writer.append("];\n");
279        }
280
281        for (Map.Entry<GraphVizEdge.Key, GraphVizEdge> e : edges.entrySet()) {
282            // GraphVizEdge.Key key = e.getKey();
283            GraphVizEdge edge = e.getValue();
284
285            writer.append("\t").append(edge.getSourceId()).append(" -> ").append(edge.getTargetId()).append(" [");
286            boolean first = true;
287            first = append(writer, "color", edge.getColor(), true, first);
288            first = append(writer, "style", edge.getStyle(), true, first);
289            first = append(writer, "arrowhead", edge.getHeadShape(), true, first);
290            first = append(writer, "arrowtail", edge.getTailShape(), true, first);
291            first = append(writer, "label", edge.getLabel(), true, first);
292            first = append(writer, "headlabel", edge.getHeadLabel(), true, first);
293            first = append(writer, "taillabel", edge.getTailLabel(), true, first);
294            first = append(writer, "lhead", edge.getLogicalHead(), false, first);
295            first = append(writer, "ltail", edge.getLogicalTail(), false, first);
296            writer.append("];\n");
297        }
298
299        /*
300         for (GraphVizCluster cluster : clusters.values()) {
301         writer.append("\tsubgraph ").append(cluster.getId()).append(" {\n");
302         if (cluster.getLabel() != null)
303         writer.append("\t\t").append("label=\"").append(String.valueOf(cluster.getLabel())).append("\";\n");
304         for (GraphVizNode.Key key : cluster.getNodeKeys()) {
305         GraphVizNode node = node(key.getScope(), key.getObject());
306         writer.append("\t\t").append(node.getId()).append(";\n");
307         }
308         writer.append("\t}\n");
309         }
310         */
311        Map<GraphVizCluster, List<GraphVizCluster>> clusterMap = newClusterMap();
312        writeTo(writer, clusterMap, null, 0);
313        writer.write("}\n");
314        writer.flush();
315    }

317    public void writeTo(@Nonnull OutputStream out) throws IOException {
318        writeTo(new OutputStreamWriter(out, Charsets.UTF_8));
319    }

321    public void writeTo(@Nonnull File file) throws IOException {
322        ByteSink sink = Files.asByteSink(file);
323        OutputStream out = sink.openBufferedStream();
324        try {
325            writeTo(out);
326        } finally {
327            out.close();
328        }
329    }
